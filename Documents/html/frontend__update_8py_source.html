105def fetch_openweather_data(city):
106 """!
107 @brief Fetch current weather data from OpenWeather API.
108 @param city City name as string.
109 @return dict: Parsed JSON response containing temperature, humidity, wind, and description.
110 """
111 try:
112 ow_url = f"{OPENWEATHER_API}?q={city}&appid={OPENWEATHER_KEY}&units=metric"
113 response = requests.get(ow_url)
114 if response.status_code == 200:
115 return response.json()
116 else:
117 st.error("Failed to fetch OpenWeather data.")
118 return {}
119 except Exception as e:
120 st.error(f"Error fetching weather: {e}")
121 return {}
